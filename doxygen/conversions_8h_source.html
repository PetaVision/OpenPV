269 static inline int nearby_neighbor(int kzPre, int zScaleLog2Pre, int zScaleLog2Post)
270 {
271  float a = powf(2.0f, (float) (zScaleLog2Pre - zScaleLog2Post));
272  int ia = (int) a;
273 
274  int k0 = (ia < 2) ? 0 : ia/2 - 1;
275 
276  // negative kzPre is different if density of post-synaptic layer decreases
277  int k = (a < 1.0f && kzPre < 0) ? kzPre - (int) (1.0f/a) + 1 : kzPre;
278 
279  return k0 + (int) (a * k);
280 }

438 static inline float deltaWithPBC(float x1, float x2, float max)
439 {
440  float dx = x2 - x1;
441  float abs_dx = fabsf(dx);
442 
443  // Apply periodic boundary conditions
444  dx = abs_dx > max ? sign(dx) * (abs_dx - 2.0f*max) : dx;
445 
446  return dx;
447 }

471 //Converts an index from one layer to the other in the extended space
472 //Warning: function will return center point in a one to many conversion
473 //Conversion in feature space does not exist, output will be first feature
474 //If outside the area of out layer, will move to the clostest avaliable position in out layer
475 static inline int layerIndexExt(int kPreExt, const PVLayerLoc * inLoc, const PVLayerLoc * outLoc){
476  //Calculate scale factor based on restricted
477  float scaleFactorX = (float)outLoc->nxGlobal / inLoc->nxGlobal;
478  float scaleFactorY = (float)outLoc->nyGlobal / inLoc->nyGlobal;
479  //Calculate x and y in extended space
480  int kPreX = kxPos(kPreExt, inLoc->nx + inLoc->halo.lt + inLoc->halo.rt, inLoc->ny+ inLoc->halo.dn + inLoc->halo.up, inLoc->nf);
481  int kPreY = kyPos(kPreExt, inLoc->nx + inLoc->halo.lt + inLoc->halo.rt, inLoc->ny+ inLoc->halo.dn + inLoc->halo.up, inLoc->nf);
482  //Subtract margin to set 0 to the beginning of the restricted space
483  kPreX -= inLoc->halo.lt;
484  kPreY -= inLoc->halo.up;
485  int kPostX, kPostY, half;
486  //If one to many, scale factor is greater than 1
487  if (scaleFactorX > 1){
488  half = floor(scaleFactorX / 2);
489  kPostX = kPreX * scaleFactorX + half;
490  }
491  else{
492  kPostX = floor(kPreX * scaleFactorX);
493  }
494  if (scaleFactorY > 1){
495  half = floor(scaleFactorY / 2);
496  kPostY = kPreY * scaleFactorY + half;
497  }
498  else{
499  kPostY = floor(kPreY * scaleFactorY);
500  }
501 
502  //Change back to ext points
503  kPostX += outLoc->halo.lt;
504  kPostY += outLoc->halo.up;
505 
506  //If outside of out layer margins, shrink
507  //Left margin
508  if (kPostX < 0){
509  kPostX = 0;
510  }
511  //Right Margin
512  else if(kPostX >= outLoc->nx + outLoc->halo.lt + outLoc->halo.rt){
513  kPostX = outLoc->nx + outLoc->halo.dn + outLoc->halo.up - 1;
514  }
515  //Top margin
516  if (kPostY < 0){
517  kPostY = 0;
518  }
519  //Bottom Margin
520  else if(kPostY >= outLoc->ny + outLoc->halo.lt + outLoc->halo.rt){
521  kPostY = outLoc->ny + outLoc->halo.dn + outLoc->halo.up - 1;
522  }
523  //Change back to index
524  //Using feature of 0
525  return kIndex(kPostX, kPostY, 0, outLoc->nx + outLoc->halo.lt + outLoc->halo.rt, outLoc->ny + outLoc->halo.dn + outLoc->halo.up, outLoc->nf);
526 }

539 //Converts a local ext index into a global res index

541 static inline int localExtToGlobalRes(int localExtK, const PVLayerLoc * loc){
542  //Change local ext indicies to global res index
543  int localExtX = kxPos(localExtK, loc->nx+loc->halo.lt+loc->halo.rt, loc->ny+loc->halo.up+loc->halo.dn, loc->nf);
544  int localExtY = kyPos(localExtK, loc->nx+loc->halo.lt+loc->halo.rt, loc->ny+loc->halo.up+loc->halo.dn, loc->nf);
545  int localF = featureIndex(localExtK, loc->nx+loc->halo.lt+loc->halo.rt, loc->ny+loc->halo.up+loc->halo.dn, loc->nf);
546 
547  //Check if restricted
548  if(localExtX < loc->halo.lt || localExtX >= loc->nx+loc->halo.lt ||
549  localExtY < loc->halo.up || localExtY >= loc->ny+loc->halo.up){
550  return -1;
551  }
552 
553  //Change ext to res
554  int globalResX = localExtX - loc->halo.lt + loc->kx0;
555  int globalResY = localExtY - loc->halo.up + loc->ky0;
556 
557  //Get final globalResK
558  int globalResK = kIndex(globalResX, globalResY, localF, loc->nxGlobal, loc->nyGlobal, loc->nf);
559  return globalResK;
560 }
