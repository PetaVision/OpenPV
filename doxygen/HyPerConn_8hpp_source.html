1098  static inline int computeMargin(int prescale, int postscale, int patchsize) {
1099  // 2^prescale is the distance between adjacent neurons in pre-layer, thus a smaller prescale means a layer with more neurons
1100  int margin = 0;
1101  if (prescale==postscale) {
1102  assert(patchsize%2==1);
1103  margin = (patchsize-1)/2;
1104  }
1105  else if (prescale<postscale) { // Density of pre is greater than density of pre: many-to-one
1106  // any patchsize is permissible
1107  int densityratio = (int) powf(2.0f,(float)(postscale-prescale));
1108  assert(densityratio % 2 == 0);
1109  margin = (patchsize-1) * densityratio/2;
1110  }
1111  else
1112  {
1113  assert(prescale>postscale); // one-to-many
1114  int densityratio = (int) powf(2.0f,(float)(prescale-postscale));
1115  int numcells = patchsize/densityratio;
1116  assert(numcells*densityratio==patchsize); // For one-to-many, patchsize must be a multiple of "many".
1117  margin = numcells/2; // integer division is correct, no matter whether numcells is even or odd
1118  }
1119  return margin;
1120  }

1122  static inline int adjustedPatchDimension(int zPre, int preNeuronsPerPostNeuron, int postNeuronsPerPreNeuron, int nPost, int patchDim, int * postStartPtr, int * patchStartPtr, int * adjustedDim) {
1123  float preInPostCoords; // The location, in postsynaptic restricted coordinates, of the presynaptic cell of this patch
1124  if (postNeuronsPerPreNeuron > 1) {
1125  preInPostCoords = zPre * postNeuronsPerPreNeuron + 0.5f*(postNeuronsPerPreNeuron-1);
1126  }
1127  else if (preNeuronsPerPostNeuron > 1) {
1128  preInPostCoords = ((float) (2*zPre-(preNeuronsPerPostNeuron-1)))/((float) 2*preNeuronsPerPostNeuron);
1129  }
1130  else {
1131  preInPostCoords = (float) zPre;
1132  }
1133  float postStartf = preInPostCoords - 0.5f*patchDim; // The location, in postsynaptic restricted coordinates of the start of an interval of length nxp and center xPreInPostCoords
1134  float postStopf = preInPostCoords + 0.5f*patchDim; // The location of the end of the interval starting at xPostStartf.
1135  // Everything between xPostStartf and xPostStopf, inclusive, is in the patch.
1136  int postStart = (int) ceil(postStartf);
1137  int postStop = (int) floor(postStopf) + 1;
1138  assert(postStop-postStart==patchDim);
1139  int patchStart = 0;
1140  int patchStop = patchDim;
1141  if (postStop<0) {
1142  postStop=0;
1143  postStart=0;
1144  patchStart=0;
1145  patchStop=0;
1146  }
1147  if (postStart<0) {
1148  patchStart += -postStart;
1149  postStart = 0;
1150  }
1151  if (postStart>nPost) {
1152  postStart=nPost;
1153  postStop=nPost;
1154  patchStart=0;
1155  patchStop=0;
1156  }
1157  if (postStop>nPost) {
1158  patchStop-=(postStop-nPost);
1159  postStop = nPost;
1160  }
1161  assert(postStop-postStart==patchStop-patchStart);
1162  // calculate width of the edge-adjusted patch and perform sanity checks.
1163  int width = patchStop - patchStart;
1164  assert(width>=0 && width<=patchDim && patchStart>=0 && patchStart+width<=patchDim);
1165  *postStartPtr = postStart;
1166  *patchStartPtr = patchStart;
1167  *adjustedDim = width;
1168  return PV_SUCCESS;
1169  }
