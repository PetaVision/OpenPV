525  static int decodeChannel(int channel_code, ChannelType * channel_type) {
526  int status = PV_SUCCESS;
527  switch( channel_code ) {
528  case CHANNEL_EXC:
529  *channel_type = CHANNEL_EXC;
530  break;
531  case CHANNEL_INH:
532  *channel_type = CHANNEL_INH;
533  break;
534  case CHANNEL_INHB:
535  *channel_type = CHANNEL_INHB;
536  break;
537  case CHANNEL_GAP:
538  *channel_type = CHANNEL_GAP;
539  break;
540  case CHANNEL_NORM:
541  *channel_type = CHANNEL_NORM;
542  break;
543  case CHANNEL_NOUPDATE:
544  *channel_type = CHANNEL_NOUPDATE;
545  break;
546  default:
547  status = PV_FAILURE;
