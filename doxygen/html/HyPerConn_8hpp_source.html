1114  static inline int computeMargin(int prescale, int postscale, int patchsize) {
1115  // 2^prescale is the distance between adjacent neurons in pre-layer, thus a smaller prescale means a layer with more neurons
1116  int margin = 0;
1117  if (prescale<postscale) { // Density of pre is greater than density of pre
1118  assert(patchsize%2==1);
1119  int densityratio = (int) powf(2.0f,(float)(postscale-prescale));
1120  margin = ((patchsize-1)/2) * densityratio;
1121  }
1122  else
1123  {
1124  int densityratio = (int) powf(2.0f,(float)(prescale-postscale));
1125  int numcells = patchsize/densityratio;
1126  assert(numcells*densityratio==patchsize && numcells%2==1);
1127  margin = (numcells-1)/2;
1128  }
1129  return margin;
1130  }
