7 // Defines datatypes for connection weights, activity buffers, membrane potentials, and GSyn buffers.
8 //
9 // Currently, only float is supported for all these types.
10 //
11 // This file is included by OpenCL kernels as well as C and C++ source files, so should be kept to minimal preprocessor directives.
12

18 #define PV_WCAST float
19

21 #define pvadata_t float
22 #define PV_ACAST float
23 #define max_pvadata_t FLT_MAX
24 #define min_pvadata_t FLT_MIN
25

28 #define max_pvvdata_t FLT_MAX
29 #define min_pvvdata_t FLT_MIN
30 
31 // The data type for a layer's GSyn buffers
32 #define pvgsyndata_t float
33 
34 // Catch-all datatype, deprecated

41 #endif // PV_DATATYPES_H_
