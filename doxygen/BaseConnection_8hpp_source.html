530  int status = PV_SUCCESS;
531  switch( channel_code ) {
532  case CHANNEL_EXC:
533  *channel_type = CHANNEL_EXC;
534  break;
535  case CHANNEL_INH:
536  *channel_type = CHANNEL_INH;
537  break;
538  case CHANNEL_INHB:
539  *channel_type = CHANNEL_INHB;
540  break;
541  case CHANNEL_GAP:
542  *channel_type = CHANNEL_GAP;
543  break;
544  case CHANNEL_NORM:
545  *channel_type = CHANNEL_NORM;
546  break;
547  case CHANNEL_NOUPDATE:
548  *channel_type = CHANNEL_NOUPDATE;
549  break;
550  default:
551  status = PV_FAILURE;
